CREATE DATABASE IF NOT EXISTS website_db;

CREATE USER IF NOT EXISTS 'website_user'@'localhost' IDENTIFIED BY 'website_pass';
CREATE USER IF NOT EXISTS 'website_user'@'%' IDENTIFIED BY 'website_pass';

GRANT ALL PRIVILEGES ON website_db.* TO 'website_user'@'localhost';
GRANT ALL PRIVILEGES ON website_db.* TO 'website_user'@'%';
FLUSH PRIVILEGES;

USE website_db;
CREATE TABLE products (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100),
    description TEXT,
    price DECIMAL(10,2),
    likes INT DEFAULT 0
);

CREATE TABLE reviews (
    id INT AUTO_INCREMENT PRIMARY KEY,
    product_id INT,
    review TEXT,
    rating INT,
    FOREIGN KEY (product_id) REFERENCES products(id) ON DELETE CASCADE
);

CREATE TABLE contacts (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100),
    email VARCHAR(100),
    message TEXT
);
